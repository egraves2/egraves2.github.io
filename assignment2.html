Code Snippet

I coded my two LEDs so that the red LED would repeatedly decrease brightness in a loop when the button is not pressed. When it is pressed, the red LED turns off and the white LED fades in an opposite pattern- the brightness increases instead of decreases. 

            
int ledRED = 9;           // the number of the RED LED pin
int b = 0;    // how bright the LED is
const int buttonPin = 2;  // the number of the pushbutton pin
const int ledWHITE = 11;  // the number of the WHITE LED pin

int buttonState = 0;  // variable for reading the pushbutton status

// the setup routine runs once when you press reset:
void setup() {
  pinMode(ledRED, OUTPUT);  // initialize the red LED pin as an output
  pinMode(ledWHITE, OUTPUT); // initialize the white LED pin as an output
  pinMode(buttonPin, INPUT); // initialize the pushbutton pin as an input
}

// the loop routine runs over and over again forever:
void loop() {
  buttonState = digitalRead(buttonPin);  // read the state of the pushbutton value
  // check if the pushbutton is pressed. If it is, the buttonState is HIGH
  if (buttonState == HIGH) {
    // turn red LED off:
    digitalWrite(ledRED, LOW);
      // increase the brightness gradually
      for(b=0; b<=255; b++)
      {
        //  set the brightness of white LED
        analogWrite(ledWHITE, b);
        delay(10);  // delay for 10 milliseconds
      }

  } else {
    // turn white LED off:
    digitalWrite(ledWHITE, LOW);
    // decrease the brightness gradually
    for(b=255; b>=0; b--)
      {
        //  set the brightness of red LED
        analogWrite(ledRED, b);
        delay(10);  // delay for 10 milliseconds
      }
  }
}
